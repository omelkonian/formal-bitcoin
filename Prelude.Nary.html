record Focus₁_⟪_⟫_ (A : Type) (B : Type) (C : Type) : Type where
  field
    focus₁ : A → B × C
  select₁ = proj₁ ∘ focus₁
  drop₁   = proj₂ ∘ focus₁

-- flipping
flip₀ : (A → B → R) → (B → A → R)
flip₀ = flip

flip₁ : (A → B → C → R) → (A → C → B → R)
flip₁ f = flip₀ ∘ f

flip₂ : (A → B → C → D → R) → (A → B → D → C → R)
flip₂ f = flip₁ ∘ f

private
  _ : flip₀ _∸_ 1 2 ≡ 1
  _ = refl

  _ : flip₀ (λ x y z → (x ∸ y) + z) 1 2 0 ≡ 1
  _ = refl

  _ : flip₁ (λ x y z → x + (y ∸ z)) 0 1 2 ≡ 1
  _ = refl

  _ : flip₁ (λ x y z w → x + (y ∸ z) + w) 0 1 2 0 ≡ 1
  _ = refl

  _ : flip₂ (λ x y z w → x + y + (z ∸ w)) 0 0 1 2 ≡ 1
  _ = refl
